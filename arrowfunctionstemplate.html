This function takes two strings and returns true if b is a palindrome of a

Named Function

            function arePalindromes(string1, string2) {
                string1 = string1.toLowerCase().replace(/\s/g, '');
                string2 = string2.toLowerCase().replace(/\s/g, '');

                if (string1.length !== string2.length) {
                  return false;
                }

                var reversedString2 = string2.split('').reverse().join('');

                return string1 === reversedString2;
              }
